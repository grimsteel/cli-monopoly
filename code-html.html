  11 │     unsigned long numTrailingZeroes = 0;
  12 │     if (_BitScanForward(&numTrailingZeroes, value)) {

  21 │     if (_BitScanReverse(&numLeadingZeroes, value)) {

   7 │ bool showYesNoPrompt(WINDOW* win, BoardState* boardState, std::string prompt, int x, int y) {
   8 │   unsigned char result = 1;
   9 │   mvwaddstr(win, y, x, prompt.c_str());
  10 │   while (true) {
  11 │     wattron(win, COLOR_PAIR(BGT_BLACK + result)); // If result is 1, then this makes it white on black
  12 │     mvwaddstr(win, y, x + static_cast<int>(prompt.length()) + 1, "Yes");
  13 │     wattroff(win, COLOR_PAIR(BGT_BLACK + result));
  14 │     wattron(win, COLOR_PAIR(BGT_WHITE - result)); // If result is 1, this makes it black on white
  15 │     mvwaddstr(win, y, x + static_cast<int>(prompt.length()) + 5, "No");
  16 │     wattroff(win, COLOR_PAIR(BGT_WHITE - result));
  17 │     wrefresh(win);
  18 │     
  19 │     int ch = wgetch(win);
  20 │     boardState->handleCharInput(ch);
  21 │     if (ch == KEY_ENTER || ch == '\n') break;
  22 │     else if (ch == KEY_LEFT || ch == KEY_RIGHT) result = !result;
  23 │   }
  24 │   wmove(win, y, x);
  25 │   wclrtoeol(win);
  26 │ 
  27 │   return result == 1;
  28 │ }

 190 │ bool BoardState::doTurn(unsigned char playerId) {
 191 │   Player* player = &players[playerId];
 192 │   unsigned char numRolls = 0;
 193 │   bool canRollAgain = true;
 194 │ 
 195 │   if (player->boardItemIndex == 255) {
 196 │     canRollAgain = false;
 197 │     player->turnsInJail++;
 198 │     // TODO: ways of leaving jail - doubles, pay, get out free
 199 │ 
 200 │     drawHeader(playerId, "Jail");
 201 │     bool wasReleased = promptJailChoices(playerId);
 202 │     if (wasReleased) {
 203 │       player->turnsInJail = 0;
 204 │       canRollAgain = true;
 205 │       numRolls = 1; // don't speed right after getting out of jail
 206 │     }
 207 │   } else {
 208 │     drawHeader(playerId, boardItems[player->boardItemIndex]->name);
 209 │   }

 211 │ 
 212 │   while (true) {
 213 │     char result = drawMenu(canRollAgain);
 214 │     switch (result) {

 217 │         unsigned char rolls = rollDice(playerId);

 220 │         unsigned char roll2 = rolls >> 3;
 221 │         unsigned char total = roll1 + roll2;
 222 │ 
 223 │         boardCenter.showDiceRoll(roll1, roll2);
 224 │ 
 225 │         // If they rolled a double, they can roll again
 226 │         canRollAgain = roll1 == roll2;
 227 │ 
 228 │         // Remove the player from the previous board item
 229 │         boardItems[player->boardItemIndex]->handlePlayerLeave(playerId);
 230 │ 
 231 │         // Get the new board item
 232 │         constexpr unsigned char NUM_BOARD_ITEMS = static_cast<unsigned char>(sizeof(boardItems) / sizeof(BoardItem*));
 233 │         unsigned char newBoardItemIndex = player->boardItemIndex + total;
 234 │ 
 235 │         // If they passed Go,
 236 │         if (newBoardItemIndex >= NUM_BOARD_ITEMS) {
 237 │           // Give the salary

 239 │           // Wrap around
 240 │           newBoardItemIndex -= NUM_BOARD_ITEMS;
 241 │         }
 242 │ 
 243 │         // If they rolled doubles three times in a row or landed on "Go to Jail", go to jail. 
 244 │         if ((canRollAgain && numRolls == 3) || newBoardItemIndex == 30) {
 245 │           canRollAgain = false;
 246 │           newBoardItemIndex = 255; // jail
 247 │         }
 248 │ 
 249 │         player->boardItemIndex = newBoardItemIndex;
 250 │         
 251 │         if (newBoardItemIndex == 255) {
 252 │           drawHeader(playerId, "Jail");
 253 │           jail.arrestPlayer(player, this);
 254 │         }
 255 │         else {
 256 │           // Draw the header, then notify the board item
 257 │           drawHeader(playerId, boardItems[player->boardItemIndex]->name);
 258 │ 
 259 │           BoardItem::RollInfo rollInfo = {
 260 │             .roll = total,
 261 │             .isChanceMultiplied = false
 262 │           };
 263 │           boardItems[player->boardItemIndex]->handlePlayer(player, this, &rollInfo);
 264 │         }
 265 │ 
 266 │         doupdate();
 267 │         break;
 268 │       } case 1: {

 273 │ 
 274 │         // Bitset (one bit per c group) indicating if all properties have the same # of houses, meaning the player can buy & sell o
     │ n all of them
 275 │         uint8_t allHousesEqual = 0b11111111;
 276 │ 
 277 │         // Calculate color groups where all properties are owned
 278 │         for (auto currentProperty = player->properties.begin(); currentProperty != player->properties.end(); currentProperty++) {
 279 │           // Skip utilities, railroads, and mortgaged properties
 280 │           if ((*currentProperty)->colorGroup == BGT_BLACK || (*currentProperty)->getHouses() == 255) continue;
 281 │ 
 282 │           int colorGroup = (*currentProperty)->colorGroup - BGT_PURPLE;

 284 │         }

 286 │         vector<unsigned char> availableColorGroups;
 287 │         availableColorGroups.reserve(sizeof(propertiesRequired) / sizeof(unsigned char));

 289 │           if (propertiesRequired[i] == 0) availableColorGroups.push_back(i);
 290 │         }
 291 │         availableColorGroups.shrink_to_fit();
 292 │ 
 293 │         if (availableColorGroups.size() == 0) break;
 294 │ 
 295 │         vector<Property*> colorGroupProperties = promptChooseProperty(availableColorGroups, true, 255);
 296 │         if (colorGroupProperties.size() == 0) break;

 299 │         cchar_t propertyDotChar;
 300 │         setcchar(&propertyDotChar, L"\uf10c", 0, colorGroup + TXT_PURPLE - BGT_PURPLE, NULL);

 302 │         setcchar(&selectedPropertyDotChar, L"\uf111", 0, colorGroup + TXT_PURPLE - BGT_PURPLE, NULL);
 303 │ 
 304 │         unsigned char selectedProperty = 0;
 305 │         vector<unsigned char> newHouses(colorGroupProperties.size());
 306 │ 
 307 │         for (int i = 0; i < colorGroupProperties.size(); i++) {
 308 │           newHouses[i] = colorGroupProperties[i]->getHouses();
 309 │         }

 311 │         short totalNewHouses = 0;

 315 │

 321 │           if (result == NavigateListResult::Cancel) break;

 328 │               }

 339 │           // Left/Right

 341 │             // Remove one house and update stats
 342 │             newHouses[selectedProperty]--;
 343 │             totalNewHouses--;
 344 │             isValidBuild = updateManageHousesStats(totalNewHouses, buildingPrice, currentMoney, newHouses);
 345 │           } else if (result == NavigateListResult::Right && newHouses[selectedProperty] < 5) {

 349 │           }
 350 │         }
 351 │         break;
 352 │       } case 2: {
 353 │           // Mortgage/unmortgage properties
 354 │           // Mortagable color groups cannot have any houses on them.
 355 │           unordered_set<unsigned char> availableColorGroupsSet;
 356 │ 
 357 │           for (auto currentProperty = player->properties.begin(); currentProperty != player->properties.end(); currentProperty++) {
 358 │             // Skip utilities/railroads
 359 │             if ((*currentProperty)->colorGroup == BGT_BLACK) continue;
 360 │ 
 361 │             int colorGroup = (*currentProperty)->colorGroup - BGT_PURPLE;
 362 │             unsigned char numHouses = (*currentProperty)->getHouses();
 363 │             bool isInSet = availableColorGroupsSet.contains(colorGroup);
 364 │             // All houses on the color group cannot have any houses (build evenly)

 366 │ 
 367 │             if (cannotBeMortgaged && isInSet) {
 368 │               availableColorGroupsSet.erase(colorGroup);

 372 │             }
 373 │           }
 374 │ 
 375 │           if (availableColorGroupsSet.size() == 0) break;
 376 │ 
 377 │           // Convert to vector
 378 │           vector<unsigned char> availableColorGroups = vector(availableColorGroupsSet.begin(), availableColorGroupsSet.end());
 379 │ 
 380 │           // Have the player choose a property
 381 │           vector<Property*> selectedProperties = promptChooseProperty(availableColorGroups, false, playerId);
 382 │           if (selectedProperties.size() == 0) break;
 383 │           Property* selectedProperty = selectedProperties[0];
 384 │ 
 385 │           short mortgageValue = selectedProperty->prices.price / 2;
 386 │ 
 387 │           // 255 house count means mortgaged
 388 │           if (selectedProperty->getHouses() == 255) {
 389 │               // Note: We need to make sure they have enough money
 390 │               short unmmortgagePrice = (mortgageValue * 11) / 10;
 391 │               if (player->getBalance() >= unmmortgagePrice) {
 392 │                   selectedProperty->setHouses(0);
 393 │ 
 394 │                   char unmortgagedCString[100];

 398 │               }
 399 │           } else {
 400 │               selectedProperty->setHouses(255);
 401 │ 
 402 │               char mortgagedCString[100];
 403 │               snprintf(mortgagedCString, 100, "Mortgaged %s", selectedProperty->name.c_str());
 404 │ 
 405 │               player->alterBalance(mortgageValue, string(mortgagedCString));
 406 │           }
 407 │           doupdate();
 408 │           break;
 409 │       } case 3: {
 410 │         // List all properties
 411 │         
 412 │         vector<Property*> selectedProperties = promptChooseProperty({ 0, 1, 2, 3, 4, 5, 6, 7 }, false, 255); // all color Groups
 413 │         if (selectedProperties.size() == 0) break;
 414 │         Property* selectedProperty = selectedProperties[0];
 415 │ 
 416 │         // Show the property details
 417 │         drawSubheader(selectedProperty->name);
 418 │ 
 419 │         wprintw(
 420 │           win,
 421 │           "Price: $%d\nMortgage Value: $%d\nBuilding Price: $%d\nRent:\n   $%d\n1  $%d\n2  $%d\n3  $%d\n4  $%d\n   $%d\n",
 422 │           selectedProperty->prices.price, selectedProperty->prices.price / 2, selectedProperty->prices.buildingPrice,
 423 │           selectedProperty->prices.rent[0], selectedProperty->prices.rent[1], selectedProperty->prices.rent[2],
 424 │           selectedProperty->prices.rent[3], selectedProperty->prices.rent[4], selectedProperty->prices.rent[5]
 425 │         );
 426 │         if (selectedProperty->getHouses() == 5) waddstr(win, "Has Hotel\n");
 427 │         else if (selectedProperty->getHouses() == 255) waddstr(win, "Mortgaged\n");
 428 │         else if (selectedProperty->getHouses() > 0) wprintw(win, "%d Houses\n", selectedProperty->getHouses());
 429 │

 431 │           cchar_t ownerChar;
 432 │           setcchar(&ownerChar, L"\uf4ca", 0, players[selectedProperty->ownedBy].color, NULL);
 433 │           wadd_wch(win, &ownerChar);

 441 │         wrefresh(win);
 442 │         wgetch(win);
 443 │         break;
 444 │       } case 4: {
 445 │         return true;
 446 │       } case 5: {
 447 │         return false;
 448 │       }
 449 │     }
 450 │   }
